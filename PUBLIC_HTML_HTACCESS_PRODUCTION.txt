# =============================================================================
# Ripe Home - Production .htaccess (public_html root)
# Bu dosyayı public_html/.htaccess olarak kaydedin
# =============================================================================

<IfModule mod_rewrite.c>
    RewriteEngine On
    
    # HTTPS Yönlendirme (zorunlu)
    RewriteCond %{HTTPS} off
    RewriteRule ^(.*)$ https://%{HTTP_HOST}%{REQUEST_URI} [L,R=301]
    
    # www olmadan kullan (opsiyonel, tercihe göre)
    # RewriteCond %{HTTP_HOST} ^www\.(.*)$ [NC]
    # RewriteRule ^(.*)$ https://%1/$1 [R=301,L]
    
    # Backend API istekleri - /api ile başlayanlar
    RewriteCond %{REQUEST_URI} ^/api/
    RewriteRule ^(.*)$ backend/public/$1 [L,QSA]
    
    # Storage istekleri - /storage ile başlayanlar (görseller)
    RewriteCond %{REQUEST_URI} ^/storage/
    RewriteRule ^(.*)$ backend/public/$1 [L,QSA]
    
    # Admin panel istekleri - /admin ile başlayanlar
    RewriteCond %{REQUEST_URI} ^/admin
    RewriteRule ^(.*)$ backend/public/$1 [L,QSA]
    
    # Filament assets istekleri
    RewriteCond %{REQUEST_URI} ^/filament/
    RewriteRule ^(.*)$ backend/public/$1 [L,QSA]
    
    # CSS ve JS assets istekleri
    RewriteCond %{REQUEST_URI} ^/css/
    RewriteRule ^(.*)$ backend/public/$1 [L,QSA]
    
    RewriteCond %{REQUEST_URI} ^/js/
    RewriteRule ^(.*)$ backend/public/$1 [L,QSA]
    
    # Frontend için - React Router (index.html'e yönlendir)
    # Eğer dosya veya klasör değilse
    RewriteCond %{REQUEST_FILENAME} !-f
    RewriteCond %{REQUEST_FILENAME} !-d
    # API, storage, admin istekleri değilse
    RewriteCond %{REQUEST_URI} !^/api/
    RewriteCond %{REQUEST_URI} !^/storage/
    RewriteCond %{REQUEST_URI} !^/admin
    RewriteCond %{REQUEST_URI} !^/backend/
    RewriteRule ^(.*)$ index.html [L,QSA]
</IfModule>

# Security Headers
<IfModule mod_headers.c>
    # XSS Protection
    Header set X-Content-Type-Options "nosniff"
    Header set X-Frame-Options "SAMEORIGIN"
    Header set X-XSS-Protection "1; mode=block"
    
    # Referrer Policy
    Header set Referrer-Policy "strict-origin-when-cross-origin"
    
    # Content Security Policy (ihtiyaca göre özelleştirin)
    # Header set Content-Security-Policy "default-src 'self' https:; img-src 'self' https: data:; script-src 'self' 'unsafe-inline' 'unsafe-eval'; style-src 'self' 'unsafe-inline';"
</IfModule>

# MIME Types
AddType image/jpeg .jpg .jpeg
AddType image/png .png
AddType image/gif .gif
AddType image/webp .webp
AddType image/svg+xml .svg
AddType application/javascript .js
AddType text/css .css
AddType application/json .json

# Gzip Compression
<IfModule mod_deflate.c>
    # Compress HTML, CSS, JavaScript, JSON, XML
    AddOutputFilterByType DEFLATE text/html
    AddOutputFilterByType DEFLATE text/css
    AddOutputFilterByType DEFLATE text/javascript
    AddOutputFilterByType DEFLATE application/javascript
    AddOutputFilterByType DEFLATE application/x-javascript
    AddOutputFilterByType DEFLATE text/xml
    AddOutputFilterByType DEFLATE application/xml
    AddOutputFilterByType DEFLATE application/json
    AddOutputFilterByType DEFLATE text/plain
</IfModule>

# Browser Caching
<IfModule mod_expires.c>
    ExpiresActive On
    
    # Images
    ExpiresByType image/jpeg "access plus 1 year"
    ExpiresByType image/jpg "access plus 1 year"
    ExpiresByType image/png "access plus 1 year"
    ExpiresByType image/gif "access plus 1 year"
    ExpiresByType image/webp "access plus 1 year"
    ExpiresByType image/svg+xml "access plus 1 year"
    ExpiresByType image/x-icon "access plus 1 year"
    
    # CSS and JavaScript
    ExpiresByType text/css "access plus 1 month"
    ExpiresByType application/javascript "access plus 1 month"
    ExpiresByType text/javascript "access plus 1 month"
    
    # HTML and JSON
    ExpiresByType text/html "access plus 0 seconds"
    ExpiresByType application/json "access plus 0 seconds"
    
    # Fonts
    ExpiresByType font/woff "access plus 1 year"
    ExpiresByType font/woff2 "access plus 1 year"
    ExpiresByType application/font-woff "access plus 1 year"
    ExpiresByType application/font-woff2 "access plus 1 year"
</IfModule>

# Cache-Control Headers
<IfModule mod_headers.c>
    # Images - 1 year
    <FilesMatch "\.(jpg|jpeg|png|gif|webp|svg|ico)$">
        Header set Cache-Control "max-age=31536000, public"
    </FilesMatch>
    
    # CSS and JavaScript - 1 month
    <FilesMatch "\.(css|js)$">
        Header set Cache-Control "max-age=2592000, public"
    </FilesMatch>
    
    # HTML and JSON - No cache
    <FilesMatch "\.(html|json)$">
        Header set Cache-Control "no-cache, no-store, must-revalidate"
        Header set Pragma "no-cache"
        Header set Expires "0"
    </FilesMatch>
</IfModule>

# Protect sensitive files
<FilesMatch "^\.">
    Order allow,deny
    Deny from all
</FilesMatch>

<FilesMatch "(?i:^\.(htaccess|htpasswd)$|^(composer\.json|composer\.lock|package\.json|package-lock\.json)$)">
    Order allow,deny
    Deny from all
</FilesMatch>

# Prevent directory browsing
Options -Indexes

# Follow symbolic links
Options +FollowSymLinks

# PHP settings (if needed)
<IfModule mod_php.c>
    php_value upload_max_filesize 20M
    php_value post_max_size 20M
    php_value max_execution_time 300
    php_value max_input_time 300
</IfModule>
